**** I made the assumption that the monster description save file will be present as well as the object save file ***

/*
 * Function: print_objects
 * --------------------------------------------------------------------------
 *  prints the 10 randomly generated objects placed in the dungeon
 */

 /*
 * Function: gen_objects
 * --------------------------------------------------------------------------
 *  generates objects from their descriptions and places them in the dungeon
 */

 /*
 * Function: destroy_objects
 * --------------------------------------------------------------------------
 * deletes all of the objects stored in the dungeon
 */

 /*
 * Function: create_object
 * --------------------------------------------------------------------------
 *  factory for createing objects from their descriptions (rolls the dice as well)
 */
